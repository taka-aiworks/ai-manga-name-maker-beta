```javascript
generateDetailedPrompt(character) {
  const settings = getCharacterSettings(character.characterId);
  const parts = [
    settings.basePrompt,        // プロンプトメーカーからの基本
    character.expression,       // 表情
    character.action,          // 動作
    character.direction,       // 向き
    character.eyes,           // 目
    character.mouth,          // 口
    character.hands           // 手
  ].filter(Boolean);
  
  return parts.join(', ');     // 完全な統合プロンプト
}
```
